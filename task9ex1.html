#empirical function of distribution F*, CDF F_0:
def D_n(Y,th):
    F = []
    for i in range(len(Y)):
        c=0
        for j in range(len(Y)):
            if (Y[j]<=Y[i]):
                c+=1
        F.append(c/len(Y) - exp.cdf(Y[i],scale=th))
    Dn = np.max(F) 
    return Dn
